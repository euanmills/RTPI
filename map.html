<!DOCTYPE html>

<html>

<head>

    <title>Interactive GeoJSON Map</title>

    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.8.0/dist/leaflet.css" />

    <style>

        #map { height: 400px; }

    </style>

</head>

<body>


<div id="map"></div>


<script src="https://unpkg.com/leaflet@1.8.0/dist/leaflet.js"></script>

<script>

    // Initialize the map centered on London

    var map = L.map('map').setView([51.5074, -0.1278], 13);


    // Load a tile layer

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

        maxZoom: 19,

        attribution: '© OpenStreetMap contributors'

    }).addTo(map);


    // Load GeoJSON from an external URL

    fetch('path_to_your_geojson_file.geojson')

        .then(function(response) {

            return response.json();

        })

        .then(function(data) {

            L.geoJson(data, {

                onEachFeature: function (feature, layer) {

                    // Assuming 'name' is a property in your GeoJSON features

                    layer.bindPopup(feature.properties.name || 'No metadata');

                }

            }).addTo(map);

        });

</script>


</body>

</html>
